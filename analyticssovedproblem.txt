        // âœ… Fetch Data and Display
        async function loadAnalytics() {
            try {
                const snapshot = await get(child(dbRef, "People"));
                const tableBody = document.querySelector("#recordsTable tbody");

                if (snapshot.exists()) {
                    const data = snapshot.val();
                    const records = Object.values(data);
                    document.getElementById("totalRecords").innerText = records.length;
                    document.getElementById("totalUsers").innerText = new Set(records.map(r => r.Name)).size;

                    // Sort by createdAt (latest first)
                    records.sort((a, b) => (b.createdAt || 0) - (a.createdAt || 0));
                    const recent = records.slice(0, 3);

                    // âœ… Table
                    tableBody.innerHTML = recent.map(r => `
                        <tr>
                            <td>${r.ID}</td>
                            <td>${r.Name || "-"}</td>
                            <td>${r.Age || "-"}</td>
                            <td>${r.createdAt ? new Date(r.createdAt).toLocaleString() : "-"}</td>
                        </tr>
                    `).join("");

                    // âœ… Chart by Age
                    const ctx = document.getElementById('recordsChart').getContext('2d');
                    new Chart(ctx, {
                        type: 'bar',
                        data: {
                            labels: records.map(r => r.Name || "Unknown"),
                            datasets: [{
                                label: 'Age',
                                data: records.map(r => parseInt(r.Age) || 0),
                                backgroundColor: 'rgba(0, 255, 200, 0.6)'
                            }]
                        },
                        options: {
                            responsive: true,
                            scales: {
                                y: { beginAtZero: true }
                            }
                        }
                    });

                } else {
                    tableBody.innerHTML = "<tr><td colspan='4'>No records found.</td></tr>";
                }

            } catch (error) {
                console.error("Error loading analytics:", error);
                alert("Failed to load analytics data.");
            }
        }

        loadAnalytics();
